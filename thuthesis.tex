        \item 模板文档类：(Xe)LaTeX 引擎处理 \texttt{thuthesis.ins} $\Rightarrow$

          \texttt{main.pdf}\\

          XeLaTeX）
        \item 可使用或参考附带的 Makefile

\begin{frame}[fragile]{论文选项}
\begin{description}
\item[bachelor] 我要写本科论文
  \begin{lstlisting}
\documentclass[bachelor]{thuthesis}
  \end{lstlisting}
\item[master] 我要写硕士论文
  \begin{lstlisting}
\documentclass[master]{thuthesis}
  \end{lstlisting}
\item[doctor] 我要写博士论文
  \begin{lstlisting}
\documentclass[doctor]{thuthesis}
  \end{lstlisting}
\item[secret] 论文有保密要求
  \begin{lstlisting}
\documentclass[doctor, secret]{thuthesis}
\secretlevel{机密}
\secretyear{2010}
  \end{lstlisting}
\end{description}
\end{frame}

\begin{frame}[fragile]{参考文献}
  \begin{itemize}
    \item 推荐 \BibTeX
      \begin{itemize}
        \item 参考文献管理自动化
        \item bib 文件
        \item bst 参考文献样式文件：\texttt{thubib.bst}
      \end{itemize}
    \item 学校要求两种引用方式：
      \begin{itemize}
        \item 上标模式：如``在许多文献$^{[12,13]}$中……''
        \begin{lstlisting}
  \cite{key12, key13}
        \end{lstlisting}
      \item 正文模式：如``文献~[14] 证明了……''
        \begin{lstlisting}
  \onlinecite{key14}
        \end{lstlisting}
      \end{itemize}
    \end{itemize}
\end{frame}
